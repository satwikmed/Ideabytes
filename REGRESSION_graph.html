models = {
    'Linear Regression': LinearRegression(),

    'Lasso Regression': Lasso(),
    'Random Forest Regression': RandomForestRegressor()
}

# Lists to store predictions and evaluation metrics
all_predictions = []
all_metrics = []
In [17]:
for model_name, model in models.items():
    print(model.fit(X_train, y_train))

    # Make predictions
    future_dates = pd.date_range(start=df['DATE'].iloc[-1], periods=60, freq='T')
    future_dates = future_dates[1:]  # Exclude the last date from the dataset
    future_dates_numeric = future_dates.astype(np.int64).values.reshape(-1, 1)  # Convert datetime to numeric values
    future_values = model.predict(future_dates_numeric)

    # Store predictions in a DataFrame
    predicted_df = pd.DataFrame({'DATE': future_dates, 'Value': future_values})
    all_predictions.append((model_name, predicted_df))


    # Calculate evaluation metrics
    y_pred = model.predict(X_test)
    mae = mean_absolute_error(y_test, y_pred)
    mse = mean_squared_error(y_test, y_pred)
    rmse = np.sqrt(mse)
    r2 = r2_score(y_test, y_pred)

    # Store metrics in a dictionary
    metrics = {
        'MAE': mae,
        'MSE': mse,
        'RMSE': rmse,
        'R2': r2
    }
    all_metrics.append((model_name, metrics))
